Below is a short, copyable Terraform root module example (provider, variables, resources, outputs); folder layout and recommended versions follow after the code.

```hcl
# main.tf
terraform {
  required_version = ">= 1.3"
  required_providers {
    azurerm = {
      source  = "hashicorp/azurerm"
      version = "~> 3.0"
    }
  }
}
provider "azurerm" {
  features {}
}

variable "prefix" { type = string; default = "myapp" }
variable "location" { type = string; default = "West Europe" }
variable "postgres_admin" { type = string; default = "pgadmin" }
variable "postgres_password" { type = string; sensitive = true }

resource "azurerm_resource_group" "rg" {
  name     = "${var.prefix}-rg"
  location = var.location
}

resource "azurerm_storage_account" "func" {
  name                     = lower(substr("${var.prefix}func",0,24))
  resource_group_name      = azurerm_resource_group.rg.name
  location                 = azurerm_resource_group.rg.location
  account_tier             = "Standard"
  account_replication_type = "LRS"
}

resource "azurerm_app_service_plan" "asp" {
  name                = "${var.prefix}-plan"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "Linux"
  reserved            = true
  sku {
    tier = "Dynamic"
    size = "Y1"
  }
}

resource "azurerm_linux_function_app" "function" {
  name                       = "${var.prefix}-func"
  resource_group_name        = azurerm_resource_group.rg.name
  location                   = azurerm_resource_group.rg.location
  service_plan_id            = azurerm_app_service_plan.asp.id
  storage_account_name       = azurerm_storage_account.func.name
  storage_account_access_key = azurerm_storage_account.func.primary_access_key
  site_config {
    linux_fx_version = "NODE|18"
  }
  app_settings = {
    FUNCTIONS_WORKER_RUNTIME = "node"
    WEBSITE_RUN_FROM_PACKAGE = "1"
  }
}

resource "azurerm_postgresql_flexible_server" "postgres" {
  name                = "${var.prefix}-pg"
  resource_group_name = azurerm_resource_group.rg.name
  location            = azurerm_resource_group.rg.location
  administrator_login          = var.postgres_admin
  administrator_login_password = var.postgres_password
  sku_name            = "Standard_B1ms"
  storage {
    storage_mb = 32768
  }
  version = "13"
}

resource "azurerm_postgresql_flexible_server_database" "appdb" {
  name                = "appdb"
  resource_group_name = azurerm_resource_group.rg.name
  server_name         = azurerm_postgresql_flexible_server.postgres.name
  charset             = "UTF8"
  collation           = "English_United States.1252"
}

# outputs.tf
output "function_default_hostname" {
  value = azurerm_linux_function_app.function.default_hostname
}

output "storage_account_name" {
  value = azurerm_storage_account.func.name
}

output "postgres_fqdn" {
  value = azurerm_postgresql_flexible_server.postgres.fqdn
}

output "postgres_admin_username" {
  value = azurerm_postgresql_flexible_server.postgres.administrator_login
}
```

Suggested folder layout: terraform/ { main.tf (or split into main.tf, variables.tf, outputs.tf), modules/ (optional reusable modules) }, and keep infra per-environment like infra/dev, infra/prod. Recommended versions: Terraform >=1.3, azurerm provider "~> 3.0" (pin to latest 3.x patch for stability); if using registry modules, pin their minor versions (e.g., v1.x) and prefer community/official Azure modules that match azurerm 3.x.
