Fetching the Terraform docs index page to find module pages and examples from dx.pagopa.it/docs/terraform.

La skill ha recuperato tutte le informazioni tramite fetch_webpage dalle pagine ufficiali su dx.pagopa.it/docs/terraform (index, using-terraform-registry-modules, code-style, required-tags, infra-folder-structure, pre-commit-terraform) e dalle guide di deployment/appsettings; qui sotto un esempio di root module che segue le convenzioni DX (tag standard, struttura file, uso di registry modules quando disponibili, e KeyVault per secret references).  

locals.tf
```
locals {
  naming_config = {
    prefix          = var.environment.prefix
    environment     = var.environment.env_short
    location        = var.environment.location
    domain          = var.environment.domain
    name            = var.environment.app_name
    instance_number = tonumber(var.environment.instance_number)
  }

  function_app_name = provider::dx::resource_name(merge(local.naming_config, { resource_type = "function_app" }))
  tags = {
    CostCenter     = "TS000 - Tecnologia e Servizi"
    CreatedBy      = "Terraform"
    Environment    = "Prod"
    BusinessUnit   = "DevEx"
    Source         = "https://github.com/pagopa/<repo>/blob/main/infra/resources/prod"
    ManagementTeam = "Developer Experience"
  }
}
```

providers.tf
```
terraform {
  required_providers {
    azurerm = { source = "hashicorp/azurerm" version = "~> 4.0" }
    dx      = { source = "pagopa-dx/azure"   version = "~> 0.0" }
  }
  backend "azurerm" {
    resource_group_name  = "<tfstate-rg>"
    storage_account_name = "<tfstate-storage>"
    container_name       = "terraform-state"
    key                  = "<project>.<layer>.<env>.tfstate"
    use_azuread_auth     = true
  }
}

provider "azurerm" { features {} storage_use_azuread = true }
provider "dx" {}
```

variables.tf
```
variable "environment" {
  type = object({
    prefix          = string
    env_short       = string
    location        = string
    domain          = optional(string)
    app_name        = string
    instance_number = string
  })
  description = "Values used to generate resource names and location short names."
}

variable "postgres_admin_username" { type = string default = "pgadmin" }
variable "postgres_admin_password" { type = string sensitive = true }
```

main.tf
```
resource "azurerm_resource_group" "this" {
  name     = "${var.environment.prefix}-${var.environment.app_name}-${var.environment.instance_number}-rg"
  location = var.environment.location
  tags     = local.tags
}

# Storage Account (example using native azurerm resource)
resource "azurerm_storage_account" "this" {
  name                     = lower("st${var.environment.prefix}${var.environment.app_name}${var.environment.instance_number}")
  resource_group_name      = azurerm_resource_group.this.name
  location                 = var.environment.location
  account_tier             = "Standard"
  account_replication_type = "LRS"
  tags                     = local.tags
}

# PostgreSQL Flexible Server (native resource; outputs in docs reference this resource type)
resource "azurerm_postgresql_flexible_server" "this" {
  name                = "${var.environment.prefix}-${var.environment.app_name}-pg-${var.environment.instance_number}"
  resource_group_name = azurerm_resource_group.this.name
  location            = var.environment.location
  sku_name            = "Standard_B1ms"
  administrator_login          = var.postgres_admin_username
  administrator_login_password = var.postgres_admin_password
  version             = "14"
  storage_mb          = 32768
  tags                = local.tags
}

# Function App: prefer registry module when available (pattern from DX registry docs)
module "function_app" {
  source  = "pagopa-dx/azure-function-app/azurerm"
  version = "~> 0.0"

  name                = local.function_app_name
  resource_group_name = azurerm_resource_group.this.name
  location            = var.environment.location

  # app_settings supports KeyVault references per DX guidance:
  # list(object({ name=string, value=optional(string), key_vault_secret_name=optional(string) }))
  app_settings = [
    { name = "WEBSITE_RUN_FROM_PACKAGE", value = "1" },
    { name = "DB_CONNECTION", key_vault_secret_name = "postgres-connection-string" }  # prefer storing connection string as KeyVault secret
  ]

  tags = local.tags
}
```

outputs.tf
```
output "resource_group" {
  value = {
    name     = azurerm_resource_group.this.name
    location = azurerm_resource_group.this.location
  }
}

output "postgres" {
  description = "Details of the PostgreSQL Flexible Server"
  value = {
    name                = azurerm_postgresql_flexible_server.this.name
    id                  = azurerm_postgresql_flexible_server.this.id
    resource_group_name = azurerm_postgresql_flexible_server.this.resource_group_name
  }
}

output "storage_account" { value = azurerm_storage_account.this.name }
```

Note rapide e riferimenti (trovati sulle pagine DX usate): usare tags centralizzate in locals.tf e passarle a tutte le risorse (Required Tags), seguire la struttura file consigliata (providers.tf, locals.tf, variables.tf, main.tf, outputs.tf) e usare i moduli del Terraform Registry con la sintassi pagopa-dx/<module>/<provider> quando disponibili (Using Registry Modules); gestire segreti via Key Vault e referenziarli come AppSettings per App Service / Function App (AppService hidden appsettings); infine abilitare i pre-commit hooks per il locking dei moduli e dei provider (.pre-commit-config.yaml) come descritto nella guida DX.
