Below is a compact, copy‑pasteable Terraform root module example (provider + vars + resources + outputs), a suggested folder layout, and recommended provider/module versions — adapt names/passwords and subnet/security to your needs.

Terraform (main.tf + variables + outputs)
```hcl
terraform {
  required_version = ">= 1.0"
  required_providers {
    azurerm = { source = "hashicorp/azurerm", version = "~> 3.0" }
    random  = { source = "hashicorp/random",  version = "~> 3.0" }
  }
}

provider "azurerm" { features {} }

resource "random_string" "suffix" { length = 6 upper = false special = false }

variable "prefix" { type = string default = "myapp" }
variable "location" { type = string default = "West Europe" }
variable "postgres_admin" { type = string default = "pgadmin" }
variable "postgres_password" { type = string description = "Set in CI/TF_VAR_postgres_password or via TF Cloud" sensitive = true }

resource "azurerm_resource_group" "rg" {
  name     = "${var.prefix}-rg"
  location = var.location
}

resource "azurerm_storage_account" "sa" {
  name                     = lower("${var.prefix}sa${random_string.suffix.result}")
  resource_group_name      = azurerm_resource_group.rg.name
  location                 = azurerm_resource_group.rg.location
  account_tier             = "Standard"
  account_replication_type = "LRS"
  allow_blob_public_access = false
}

resource "azurerm_storage_container" "function_pkg" {
  name                  = "function-pkg"
  storage_account_name  = azurerm_storage_account.sa.name
  container_access_type = "private"
}

resource "azurerm_app_service_plan" "plan" {
  name                = "${var.prefix}-plan"
  location            = azurerm_resource_group.rg.location
  resource_group_name = azurerm_resource_group.rg.name
  kind                = "Linux"
  reserved            = true
  sku {
    tier = "ElasticPremium"
    size = "EP1"
  }
}

resource "azurerm_linux_function_app" "func" {
  name                       = "${var.prefix}-func-${random_string.suffix.result}"
  resource_group_name        = azurerm_resource_group.rg.name
  location                   = azurerm_resource_group.rg.location
  service_plan_id            = azurerm_app_service_plan.plan.id
  storage_account_name       = azurerm_storage_account.sa.name
  storage_account_access_key = azurerm_storage_account.sa.primary_access_key

  site_config {
    linux_fx_version = "NODE|18"
  }

  app_settings = {
    FUNCTIONS_WORKER_RUNTIME = "node"
    WEBSITE_RUN_FROM_PACKAGE = "1"
    AzureWebJobsStorage      = azurerm_storage_account.sa.primary_connection_string
  }
}

resource "azurerm_postgresql_flexible_server" "pg" {
  name                = "${var.prefix}-pg-${random_string.suffix.result}"
  resource_group_name = azurerm_resource_group.rg.name
  location            = azurerm_resource_group.rg.location
  administrator_login = var.postgres_admin
  administrator_login_password = var.postgres_password
  sku_name            = "Standard_B1ms"
  storage_mb          = 32768
  version             = "13"
  backup_retention_days = 7
  high_availability   = "Disabled"
  public_network_access_enabled = true
}

output "function_default_hostname" {
  value = azurerm_linux_function_app.func.default_hostname
}

output "storage_account_name" {
  value = azurerm_storage_account.sa.name
}

output "postgresql_fqdn" {
  value = azurerm_postgresql_flexible_server.pg.fqdn
}
```

Suggested folder layout (minimal)
- terraform/
  - main.tf         (combine provider + resources for small project)
  - variables.tf
  - outputs.tf
  - modules/
    - function_app/  (optional: move Function-related resources here)
    - postgresql/    (optional: move DB resources here)
  - envs/
    - dev/           (overrides and backend config)
    - prod/

Recommended versions & notes
- Terraform core: >= 1.0 (use latest stable)
- azurerm provider: "~> 3.0" (adapt to your org policy; v4+ breaking changes exist)
- random provider: "~> 3.0" (for safe name suffixes)
- Modules: prefer small, reviewed modules (e.g., create local modules under modules/* or use verified registry modules pinned to specific versions)
- Security: supply postgres password via secret manager / CI (do not hardcode), restrict DB network access with subnets/private endpoints in prod.
